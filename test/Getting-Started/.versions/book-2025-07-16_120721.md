using WGLMakie
using Colors

# Create a cool animated spiral visualization
fig = Figure(resolution = (800, 800), backgroundcolor = :black)
ax = Axis(fig[1, 1], aspect = 1)
hidedecorations!(ax)
hidespines!(ax)

# Time observable for animation
t = Observable(0.0)

# Create animated spiral points
points = @lift begin
    n_points = 500
    theta = range(0, 4π, length=n_points)
    r = theta / (4π)
    
    # Add time-based animation
    phase = $t * 2
    
    # Create spiral with wave
    x = @. r * cos(theta + phase) * (1 + 0.2 * sin(5 * theta + phase))
    y = @. r * sin(theta + phase) * (1 + 0.2 * sin(5 * theta + phase))
    
    Point2f.(x, y)
end

# Create colors that change over time
colors = @lift begin
    n_points = 500
    [HSV(360 * (i/n_points + $t/10) % 360, 0.8, 0.9) for i in 1:n_points]
end

# Plot the spiral
lines!(ax, points, color = colors, linewidth = 3)

# Add a rotating scatter plot
scatter_points = @lift begin
    n = 12
    angles = range(0, 2π, length=n+1)[1:n] .+ $t
    r = 0.8
    [Point2f(r * cos(a), r * sin(a)) for a in angles]
end

scatter!(ax, scatter_points, color = :white, markersize = 20)

# Add title with animation info
title = Label(fig[0, 1], "Animated Spiral Galaxy", fontsize = 24, color = :white)

# Create animation
@async while !Makie.isclosed(fig.scene)
    t[] = t[] + 0.02
    sleep(0.03)
end

using WGLMakie

# Create a beautiful animated 3D visualization
fig = Figure(resolution = (800, 600))

# Create 3D axis
ax = Axis3(fig[1, 1], 
    title = "Animated 3D Torus Knot",
    aspect = :data
)

# Time observable for animation
time = Observable(0.0)

# Parametric equations for a torus knot
t = LinRange(0, 2π, 500)
r = 2
R = 3

# Create animated torus knot
x = @lift(@. R * cos(2*t) + r * cos(2*t) * cos(3*t + $time))
y = @lift(@. R * sin(2*t) + r * sin(2*t) * cos(3*t + $time))
z = @lift(@. r * sin(3*t + $time))

# Color based on position
colors = @lift(@. sin(3*t + 2*$time))

# Plot the torus knot
lines!(ax, x, y, z, 
    color = colors,
    colormap = :rainbow,
    linewidth = 5
)

# Add some floating spheres
n_spheres = 20
sphere_t = LinRange(0, 2π, n_spheres)
sphere_x = @lift(@. 4 * cos.(sphere_t .+ $time))
sphere_y = @lift(@. 4 * sin.(sphere_t .+ $time))
sphere_z = @lift(@. 2 * sin.(2 * sphere_t .+ 2*$time))

scatter!(ax, sphere_x, sphere_y, sphere_z,
    color = :gold,
    markersize = 15,
    transparency = true
)

# Animate
on(fig.scene.events.tick) do tick
    time[] += 0.03
end

# Display
fig
